	// Types found here:
	// http://www2.census.gov/geo/pdfs/maps-data/data/tiger/tgrshp2010/TGRSHP10SF1CH5.pdf
	// http://www.ofm.wa.gov/pop/geographic/readme_geo.pdf
	//
	// Key:
	// FP == FIPS: Federal Information Processing Standard
	// NS == GNIS: Geographic Name Information System (USGS)
	// HL: includes Hispanic and Latino
	// (m) == meters (assuming square)
	// (mi) == miles (assuming square)

	StateFP10  string  `xml:"STATEFP10"`  // State FIPS 55 code
	CountyFP10 string  `xml:"COUNTYFP10"` // County FIPS 55 code
	CountyNS10 string  `xml:"COUNTYNS10"` // County GNIS code
	GeoID10    string  `xml:"GEOID10"`    // State FIPS + County FIPS
	Name10     string  `xml:"NAME10"`     // Regional Name
	NameLSAD10 string  `xml:"NAMELSAD10"` // Legal name
	LSAD10     string  `xml:"LSAD10"`     // Legal description code
	ClassFP10  string  `xml:"CLASSFP10"`  // FIPS class code
	MTFCC10    string  `xml:"MTFCC10"`    // MAF/TIGER feature class code
	FuncStat10 string  `xml:"FUNCSTAT10"` // Census functional status
	IntplLon   string  `xml:"INTPLLON"`   // Interpolated Longiture
	IntplLat   string  `xml:"INTPLLAT"`   // Interpolated Latitude
	ALandM     float64 `xml:"ALANDM"`     // Land area (m)
	AWaterM    float64 `xml:"AWATERM"`    // Water area (m)
	ALandMI    float64 `xml:"ALANDMI"`    // Land area (mi)
	AWaterMI   float64 `xml:"AWATERMI"`   // Water area (mi)
	Pop10      int     `xml:"POP10"`      // Total population
	HHP10      int     `xml:"HHP10"`      // Household population
	GQ10       int     `xml:"GQ10"`       // Group quarters population
	HU10       int     `xml:"HU10"`       // Total housing units
	OHU10      int     `xml:"OHU10"`      // Occupied housing units
	PopWhite   int     `xml:"POPWHITE"`   // White HL
	PopBlack   int     `xml:"POPBLACK"`   // Black/African-American HL
	PopAIAN    int     `xml:"POPAIAN"`    // American Indian/Alaska Native HL
	PopAsian   int     `xml:"POPASIAN"`   // Asian HL
	PopNHoPI   int     `xml:"POPNHOPI"`   // Native Hawaiian/Pacific Is. HL
	PopOth     int     `xml:"POPOTH"`     // Some other race HL
	PopTwo     int     `xml:"POPTWO"`     // Two or more races HL
	PopHisp    int     `xml:"POPHISP"`    // Hispanic/Latino
	PopWhite2  int     `xml:"POPWHITE2"`  // White
	PopBlack2  int     `xml:"POPBLACK2"`  // Black/African-American
	PopAIAN2   int     `xml:"POPAIAN2"`   // American Indian/Alaska Native
	PopAsian2  int     `xml:"POPASIAN2"`  // Asian
	PopNHoPI2  int     `xml:"POPNHOPI2"`  // Native Hawaiian/Pacific Is.
	PopOth2    int     `xml:"POPOTH2"`    // Some other race
	PopTwo2    int     `xml:"POPTWO2"`    // Two or more races
